    ```python
    """JSON-RPC 2.0 helpers + stdio hardening for the Cyrus MCP server.

    Stdlib only. Imported by cyrus.server (main loop) and cyrus.tools
    (error shaping). Kept deliberately small so the lint gate
      grep -rE "^\\s*print\\(" src/cyrus/jsonrpc.py
    trivially stays at zero hits.
    """
    from __future__ import annotations

    import io
    import json
    import os
    import sys
    from typing import Any

    # JSON-RPC 2.0 reserved error codes
    PARSE_ERROR = -32700
    INVALID_REQUEST = -32600
    METHOD_NOT_FOUND = -32601
    INVALID_PARAMS = -32602
    INTERNAL_ERROR = -32603

    # MCP protocol versions this server accepts. Echo back whatever the
    # client sent; do NOT force-downgrade. (Negotiation lives in server.py.)
    ACCEPTED_PROTOCOL_VERSIONS = frozenset({
        "2025-11-25", "2025-03-26", "2024-11-05",
    })


    class JsonRpcError(ValueError):
        """Raised by parse() when input is malformed. `.code` is one of
        PARSE_ERROR / INVALID_REQUEST. Caught by the server loop and
        converted to an emit_error() response."""
        def __init__(self, code: int, message: str):
            super().__init__(message)
            self.code = code


    def parse(line: str) -> dict[str, Any]:
        """Parse one newline-delimited JSON-RPC message.

        Strips trailing whitespace. Raises JsonRpcError(PARSE_ERROR) on
        malformed JSON. Raises JsonRpcError(INVALID_REQUEST) on JSON-RPC
        batch arrays (unsupported in v1 per CONTEXT.md scope).
        Non-object payloads (strings, numbers, null) also map to
        INVALID_REQUEST.
        """
        stripped = line.strip()
        try:
            obj = json.loads(stripped)
        except json.JSONDecodeError as e:
            raise JsonRpcError(PARSE_ERROR, f"Parse error: {e}") from e
        if isinstance(obj, list):
            raise JsonRpcError(INVALID_REQUEST, "JSON-RPC batch not supported")
        if not isinstance(obj, dict):
            raise JsonRpcError(INVALID_REQUEST, "request must be an object")
        return obj


    def emit(stream, payload: dict[str, Any]) -> None:
        """Write payload as one newline-delimited JSON line and flush.

        Uses json.dumps with separators=(",",":") — no spaces, no embedded
        newlines (json.dumps escapes \\n inside strings by default). The
        trailing "\\n" is the framing delimiter.
        """
        line = json.dumps(payload, separators=(",", ":"), ensure_ascii=False)
        stream.write(line + "\n")
        stream.flush()


    def emit_error(
        stream, request_id: Any, code: int, message: str
    ) -> None:
        """Emit a JSON-RPC error response. request_id may be None (→ null)."""
        emit(stream, {
            "jsonrpc": "2.0",
            "id": request_id,
            "error": {"code": code, "message": message},
        })


    def harden_stdio():
        """Swap sys.stdout→sys.stderr and return a protected real-stdout
        handle that the server loop writes protocol messages to.

        Steps (in order, before ANY non-stdlib import by the caller):
          1. Capture the real stdout binary buffer.
          2. Replace sys.stdout with sys.stderr so any stray print()
             from subsequent imports or library code goes to stderr
             harmlessly (Pitfall #2 — what killed MemPalace).
          3. On Windows, put the underlying stdin/stdout file descriptors
             into binary mode via msvcrt.setmode(fd, O_BINARY) so \\r\\n
             translation cannot mangle JSON payloads (Pitfall #3).
          4. Wrap the captured stdout buffer in an io.TextIOWrapper with
             encoding="utf-8", errors="replace", newline="\\n",
             write_through=True so cp1252 defaults on Windows Python
             3.14+ cannot UnicodeEncodeError us (Pitfall #4).

        Returns the protected handle. Caller stashes it and writes every
        protocol message through it. The raw sys.stdout is now aliased
        to stderr; stray prints are absorbed.

        Not idempotent — calling twice swaps stderr back in as stdout.
        Call exactly once at server boot.
        """
        real_stdout_buffer = sys.stdout.buffer
        sys.stdout = sys.stderr  # any stray print() → stderr

        if sys.platform == "win32":
            try:
                import msvcrt  # type: ignore[import-not-found]
                msvcrt.setmode(sys.stdin.fileno(), os.O_BINARY)
                msvcrt.setmode(real_stdout_buffer.fileno(), os.O_BINARY)
            except (OSError, AttributeError, ValueError):
                # Test harnesses wrap stdio in non-fd-backed streams;
                # binary-mode is a real-process-only concern.
                pass
            # Also rewrap stdin for UTF-8 reads
            try:
                sys.stdin = io.TextIOWrapper(
                    sys.stdin.buffer, encoding="utf-8",
                    errors="replace", newline="\n",
                )
            except (AttributeError, ValueError):
                pass

        protected = io.TextIOWrapper(
            real_stdout_buffer,
            encoding="utf-8",
            errors="replace",
            newline="\n",
            write_through=True,
        )
        return protected
    ```

    ```python
    """Tool handlers for the Cyrus MCP server.

    Every handler is a pure function that delegates to existing library
    code (cyrus.storage, cyrus.search, cyrus.rules). Handlers accept
    kwargs matching the tool's inputSchema and return JSON-serializable
    dicts. Errors raise — the server loop wraps them into MCP error
    responses (isError=true) in Plan 05-02.

    No protocol logic here. No I/O outside the library calls. No print().
    Keep handlers boring — the interesting stuff lives in storage/search/
    rules and has its own tests.
    """
    from __future__ import annotations

    import os
    from datetime import datetime
    from pathlib import Path
    from typing import Any

    from cyrus.logutil import get_logger
    from cyrus.paths import CATEGORIES, category_dir, cyrus_home
    from cyrus.storage import (
        atomic_write,
        dump_frontmatter,
        parse_frontmatter,
        save_memory,
    )

    _log = get_logger(__name__)


    def cyrus_save(
        category: str,
        content: str,
        tags: list[str] | None = None,
        source: str | None = None,
    ) -> dict[str, Any]:
        """Delegates to cyrus.storage.save_memory."""
        path = save_memory(
            category=category,
            content=content,
            tags=tags,
            source=source,
        )
        # id lives in filename between the date and slug: YYYY-MM-DD_<id>_<slug>.md
        id_hex = path.stem.split("_", 2)[1]
        return {"path": str(path), "id": id_hex}


    def cyrus_search(
        query: str,
        category: str | None = None,
        limit: int = 10,
        tags: list[str] | None = None,
    ) -> dict[str, Any]:
        """Delegates to cyrus.search.search; serializes SearchResult."""
        from cyrus.search import search as _search  # lazy; search imports are heavy
        results = _search(
            query=query, category=category, limit=limit, tags=tags,
        )
        return {
            "results": [
                {
                    "path": str(r.path),
                    "score": r.score,
                    "snippet": r.snippet,
                    "metadata": r.metadata,
                }
                for r in results
            ]
        }


    def cyrus_list(
        category: str | None = None,
        limit: int = 20,
        since: str | None = None,
    ) -> dict[str, Any]:
        """List memory metadata across (or within) categories. No body."""
        if category is not None and category not in CATEGORIES:
            raise ValueError(
                f"Unknown category {category!r}. Valid: {CATEGORIES}"
            )
        roots = (
            [category_dir(category)] if category is not None
            else [cyrus_home() / c for c in CATEGORIES]
        )
        out: list[dict[str, Any]] = []
        for root in roots:
            if not root.exists():
                continue
            for p in sorted(root.glob("*.md")):
                try:
                    text = p.read_text(encoding="utf-8", errors="replace")
                    metadata, _body = parse_frontmatter(text)
                except (OSError, ValueError) as e:
                    _log.warning("cyrus_list: skipping %s: %s", p, e)
                    continue
                if since is not None:
                    updated = metadata.get("updated", "")
                    if not updated or updated < since:
                        continue
                out.append({
                    "path": str(p),
                    "category": metadata.get("category", ""),
                    "id": metadata.get("id", ""),
                    "created": metadata.get("created", ""),
                    "updated": metadata.get("updated", ""),
                    "tags": metadata.get("tags", []),
                })
        # Sort by updated desc, truncate to limit.
        out.sort(key=lambda m: m.get("updated", ""), reverse=True)
        return {"memories": out[:limit]}


    def cyrus_delete(path: str) -> dict[str, Any]:
        """Delete a memory file. Path MUST resolve under cyrus_home()."""
        try:
            target = Path(path).resolve()
            home = cyrus_home().resolve()
            # Scope check — refuse arbitrary FS access (Pitfall 22/23).
            try:
                target.relative_to(home)
            except ValueError:
                return {
                    "success": False,
                    "error": f"path not under cyrus_home(): {path}",
                }
            if not target.exists():
                return {"success": False, "error": f"not found: {path}"}
            target.unlink()
            return {"success": True, "path": str(target)}
        except OSError as e:
            return {"success": False, "error": str(e)}


    def cyrus_status() -> dict[str, Any]:
        """Counts + hook-error log length. No I/O beyond glob + wc."""
        home = cyrus_home()
        by_category: dict[str, int] = {}
        total = 0
        for cat in CATEGORIES:
            d = home / cat
            n = sum(1 for _ in d.glob("*.md")) if d.exists() else 0
            by_category[cat] = n
            total += n
        rules_dir = home / "rules"
        rules_count = sum(1 for _ in rules_dir.glob("*.md")) if rules_dir.exists() else 0
        # Hook errors: read line count of ~/.cyrus/hook-errors.log if present.
        hook_errors = 0
        err_log = home / "hook-errors.log"
        if err_log.exists():
            try:
                hook_errors = sum(1 for _ in err_log.open("r", encoding="utf-8", errors="replace"))
            except OSError:
                hook_errors = -1
        # Recent: top 5 memories by updated desc across all categories.
        recent: list[dict[str, Any]] = []
        for cat in CATEGORIES:
            d = home / cat
            if not d.exists():
                continue
            for p in d.glob("*.md"):
                try:
                    meta, _ = parse_frontmatter(
                        p.read_text(encoding="utf-8", errors="replace")
                    )
                except (OSError, ValueError):
                    continue
                recent.append({
                    "path": str(p),
                    "category": meta.get("category", cat),
                    "updated": meta.get("updated", ""),
                })
        recent.sort(key=lambda m: m.get("updated", ""), reverse=True)
        return {
            "total": total,
            "by_category": by_category,
            "rules_count": rules_count,
            "recent": recent[:5],
            "hook_errors": hook_errors,
        }


    def cyrus_add_rule(
        name: str,
        severity: str,
        matches: list[str],
        pattern: str,
        message: str,
        priority: int = 50,
        triggers: list[str] | None = None,
    ) -> dict[str, Any]:
        """Create a rule file under rules/. Validates regex BEFORE write."""
        if severity not in ("block", "warn"):
            raise ValueError(
                f"severity must be 'block' or 'warn', got {severity!r}"
            )
        if not name or "/" in name or "\\" in name:
            raise ValueError(f"invalid rule name: {name!r}")
        # Hard pre-flight: regex must compile. Delegates to the same helper
        # cyrus.rules uses, so anchoring + compile behavior is consistent.
        from cyrus._rulesutil import _compile_rule_pattern
        _compile_rule_pattern(pattern, anchored=True)  # raises re.error on bad pattern

        rules_dir = category_dir("rules")
        rules_dir.mkdir(parents=True, exist_ok=True)
        rule_path = rules_dir / f"{name}.md"
        effective_triggers = list(triggers) if triggers else ["PreToolUse"]
        metadata = {
            "name": name,
            "severity": severity,
            "triggers": effective_triggers,
            "matches": list(matches),
            "pattern": pattern,
            "priority": int(priority),
            "message": message,
        }
        document = dump_frontmatter(metadata, "")
        atomic_write(rule_path, document)
        return {"path": str(rule_path)}

        "cyrus_status":   cyrus_status,
        "cyrus_add_rule": cyrus_add_rule,
    }
    ```
