initialize request (client → server):

  {"jsonrpc":"2.0","id":1,
   "result":{"protocolVersion":"<echo client>",
             "serverInfo":{"name":"cyrus","version":"<package version>"},
             "capabilities":{"tools":{}}}}

If client sends an unknown protocolVersion: echo the server's preferred
("2025-03-26") back rather than erroring — Claude Code tolerates this.

notifications/initialized:
  Request has no "id" field. Return None (no response written).

tools/list:
  Response: {"jsonrpc":"2.0","id":N,"result":{"tools": TOOLS}}

tools/call:
  Request params: {"name":"cyrus_status","arguments":{}}
  Success response:
    {"jsonrpc":"2.0","id":N,
     "result":{"content":[{"type":"text","text":"<json.dumps(handler_return)>"}]}}
  Handler error response (MCP style, NOT JSON-RPC error):
    {"jsonrpc":"2.0","id":N,
     "result":{"content":[{"type":"text","text":"<error message>"}],
               "isError":true}}
  Unknown tool name: JSON-RPC -32601 error (not tools/call isError).

ping:
  Response: {"jsonrpc":"2.0","id":N,"result":{}}

notifications/cancelled:
  Notification (no id). Log to stderr, return None.

Unknown methods (prompts/list, resources/list, etc.):
  Return JSON-RPC error -32601. Do NOT crash.

All responses use json.dumps(..., separators=(",",":")) + "\n". Never
use f-strings to build JSON (Pitfall #12 — backslash corruption on
Windows paths).

    ```python
    """Cyrus MCP stdio server. Long-lived process invoked as `cyrus serve`.

    Transport: newline-delimited JSON-RPC 2.0. NO Content-Length framing
    (Pitfall #1 — what killed MemPalace with Claude Code).

    Stdout is sacred. We call cyrus.jsonrpc.harden_stdio() FIRST — before
    any non-stdlib import in this module that could transitively print —
    so that any stray print() ends up on stderr. All protocol writes go
    to the protected real-stdout handle returned by harden_stdio().

    Lazy-import policy: heavyweight modules (cyrus.tools, cyrus.schemas,
    cyrus.search) are imported inside main() AFTER harden_stdio() runs.
    Keeps the window during which a stray print could corrupt the stream
    vanishingly small.
    """
    from __future__ import annotations

    import json
    import sys
    from typing import Any

    # NOTE: deliberately NO top-level imports from cyrus.tools / .search /
    # .schemas / .storage — see docstring. handle_request() is unit-tested
    # with in-process imports (already safe once harden_stdio has run in
    # the test harness) but main() imports them only AFTER the swap.

    from cyrus.jsonrpc import (
        ACCEPTED_PROTOCOL_VERSIONS,
        INTERNAL_ERROR,
        INVALID_PARAMS,
        INVALID_REQUEST,
        JsonRpcError,
        METHOD_NOT_FOUND,
        PARSE_ERROR,
        emit,
        emit_error,
        harden_stdio,
        parse,
    )
    from cyrus.logutil import get_logger

    _log = get_logger(__name__)

    _PREFERRED_VERSION = "2025-03-26"


    def _server_version() -> str:
        """Return the installed cyrus package version, or '0.0.0' in dev."""
        try:
            from importlib.metadata import version, PackageNotFoundError
            try:
                return version("cyrus")
            except PackageNotFoundError:
                return "0.0.0"
        except ImportError:  # pragma: no cover — stdlib on 3.8+
            return "0.0.0"


    def _initialize(params: dict[str, Any]) -> dict[str, Any]:
        """Echo the client's protocolVersion when we know it, else preferred."""
        client_version = params.get("protocolVersion", "")
        echo = (
            client_version
            if client_version in ACCEPTED_PROTOCOL_VERSIONS
            else _PREFERRED_VERSION
        )
        return {
            "protocolVersion": echo,
            "serverInfo": {"name": "cyrus", "version": _server_version()},
            "capabilities": {"tools": {}},
        }


    def _tools_list() -> dict[str, Any]:
        from cyrus.schemas import TOOLS
        return {"tools": TOOLS}


    def _tools_call(params: dict[str, Any]) -> dict[str, Any]:
        """Dispatch to a cyrus_* handler. Handler exceptions → isError."""
        from cyrus.tools import HANDLERS
        name = params.get("name")
        arguments = params.get("arguments") or {}
        if name not in HANDLERS:
            # Unknown tool → JSON-RPC method error (spec-compliant) by
            # raising JsonRpcError; the outer handle_request converts.
            raise JsonRpcError(METHOD_NOT_FOUND, f"Unknown tool: {name}")
        try:
            result = HANDLERS[name](**arguments)
        except TypeError as e:
            # Missing/extra kwargs — treat as invalid params, not crash.
            raise JsonRpcError(INVALID_PARAMS, f"Invalid arguments for {name}: {e}") from e
        except Exception as e:  # noqa: BLE001 — surface to client as isError
            return {
                "content": [{"type": "text", "text": f"{type(e).__name__}: {e}"}],
                "isError": True,
            }
        # Success: JSON-stringify the handler result into the MCP text block.
        return {
            "content": [
                {"type": "text", "text": json.dumps(result, separators=(",", ":"))}
            ]
        }


    def handle_request(request: dict[str, Any]) -> dict[str, Any] | None:
        """Dispatch one JSON-RPC request. Returns the response dict, or
        None for notifications (no "id" field).

        Never raises: every error path returns a well-formed JSON-RPC
        error response. Used directly by the server loop AND by unit
        tests in tests/test_server.py (which is why it's a pure function
        over dicts rather than operating on streams).
        """
        method = request.get("method")
        request_id = request.get("id")
        is_notification = "id" not in request

        if not isinstance(method, str) or not method:
            if is_notification:
                return None
            return {
                "jsonrpc": "2.0",
                "id": request_id,
                "error": {"code": INVALID_REQUEST, "message": "missing method"},
            }

        params = request.get("params") or {}

        try:
            if method == "initialize":
                result: Any = _initialize(params)
            elif method == "notifications/initialized":
                return None
            elif method == "notifications/cancelled":
                _log.info(
                    "cyrus.server: notifications/cancelled requestId=%s",
                    params.get("requestId"),
                )
                return None
            elif method == "tools/list":
                result = _tools_list()
            elif method == "tools/call":
                result = _tools_call(params)
            elif method == "ping":
                result = {}
            else:
                if is_notification:
                    # Ignore unknown notifications silently per JSON-RPC 2.0.
                    return None
                return {
                    "jsonrpc": "2.0",
                    "id": request_id,
                    "error": {"code": METHOD_NOT_FOUND,
                              "message": f"Method not found: {method}"},
                }
        except JsonRpcError as e:
            if is_notification:
                return None
            return {
                "jsonrpc": "2.0",
                "id": request_id,
                "error": {"code": e.code, "message": str(e)},
            }
        except Exception as e:  # noqa: BLE001 — fail-loud but survive
            _log.exception("cyrus.server: handler crashed on %s", method)
            if is_notification:
                return None
            return {
                "jsonrpc": "2.0",
                "id": request_id,
                "error": {"code": INTERNAL_ERROR, "message": f"{type(e).__name__}: {e}"},
            }

        if is_notification:
            return None
        return {"jsonrpc": "2.0", "id": request_id, "result": result}


    def main() -> int:
        """Server entry point. Blocks on stdin until EOF or fatal error."""
        # FIRST — before any touch of cyrus.tools / .search / .schemas.
        protected_stdout = harden_stdio()

        _log.info("cyrus.server: started")
        try:
            for line in sys.stdin:
                if not line.strip():
                    continue
                try:
                    request = parse(line)
                except JsonRpcError as e:
                    emit_error(protected_stdout, None, e.code, str(e))
                    continue
                response = handle_request(request)
                if response is not None:
                    emit(protected_stdout, response)
        except KeyboardInterrupt:
            return 0
        except BrokenPipeError:
            # Claude Code closed stdin — normal shutdown.
            return 0
        return 0


    if __name__ == "__main__":
        raise SystemExit(main())
    ```

    ```python
    def _spawn_server(self, env_home: Path) -> subprocess.Popen:
        env = {**os.environ, "CYRUS_HOME": str(env_home)}
        return subprocess.Popen(
            [sys.executable, "-m", "cyrus.cli", "serve"],
            stdin=subprocess.PIPE, stdout=subprocess.PIPE,
            stderr=subprocess.PIPE, env=env,
            # Crucial: text=False so we control newline translation
            # ourselves and the Windows msvcrt.setmode binary-mode path
            # is actually exercised.
            bufsize=0,
        )

    def _rpc(self, proc, payload: dict) -> dict | None:
        line = (json.dumps(payload, separators=(",", ":")) + "\n").encode("utf-8")
        proc.stdin.write(line); proc.stdin.flush()
        if "id" not in payload:
            return None  # notification — no response expected
        raw = proc.stdout.readline()
        return json.loads(raw.decode("utf-8"))
    ```
